## 📊 数据表结构设计

### 主表：geo_monitor_results（监控结果表）

| 字段名 | 类型 | 说明 | 示例 |
|--------|------|------|------|
| **id** | INT | 自增ID | 1 |
| **question** | VARCHAR(500) | 问题/关键词 | "steam错误代码118怎么办" |
| **ai_model** | VARCHAR(50) | AI模型/平台 | 百度/豆包/DeepSeek/文心一言 |
| **target_brand** | VARCHAR(100) | 目标品牌 | 奇游加速器 |
| **ai_answer** | TEXT | AI完整回答 | [AI生成的完整文本] |
| **query_time** | DATETIME | 查询时间 | 2026-03-18 09:00:00 |
| **brand_appeared** | BOOLEAN | 是否出现品牌目标 | TRUE/FALSE |
| **position** | VARCHAR(20) | 位置/排名 | 第1位/第2位/未出现 |

| **negative_keywords** | VARCHAR(500) | 负面关键词 | 贵,卡顿,不好用 |

## 🔧 Python实现方案

### 完整代码框架

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
GEO效果监控系统
支持平台：百度、豆包、DeepSeek、文心一言
"""

import pandas as pd
import json
import os
import re
import time
from datetime import datetime
from typing import List, Dict, Optional
import schedule

# 浏览器自动化
from playwright.sync_api import sync_playwright, Page

# 数据存储
import sqlite3
from dataclasses import dataclass, asdict


@dataclass
class GEOResult:
    """GEO监控结果数据类"""
    question: str
    ai_model: str
    target_brand: str
    ai_answer: str
    query_time: datetime
    brand_appeared: bool
    position: str
    positive_keywords: str
    negative_keywords: str
    competitors_mentioned: str
    references: str
    screenshot_path: str


class GEOMonitor:
    """GEO效果监控主类"""
    
    def __init__(self, config_file: str = "geo_config.xlsx"):
        """初始化监控器"""
        self.config_file = config_file
        self.config_df = self._load_config()
        self.results = []
        
        # 创建截图目录
        self.screenshot_dir = "screenshots"
        os.makedirs(self.screenshot_dir, exist_ok=True)
        
        # 初始化数据库
        self._init_database()
    
    def _load_config(self) -> pd.DataFrame:
        """加载监控配置"""
        if os.path.exists(self.config_file):
            return pd.read_excel(self.config_file)
        else:
            # 创建示例配置
            sample_config = {
                "question": ["steam错误代码118怎么办", "加速器哪个好用"],
                "ai_model": ["百度", "豆包"],
                "target_brand": ["奇游加速器", "奇游加速器"],
                "priority": ["P0", "P1"],
                "category": ["Steam问题", "产品推荐"],
                "frequency": ["daily", "daily"],
                "status": ["active", "active"]
            }
            df = pd.DataFrame(sample_config)
            df.to_excel(self.config_file, index=False)
            return df
    
    def _init_database(self):
        """初始化SQLite数据库"""
        conn = sqlite3.connect('geo_monitor.db')
        cursor = conn.cursor()
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS geo_monitor_results (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                question TEXT,
                ai_model TEXT,
                target_brand TEXT,
                ai_answer TEXT,
                query_time TIMESTAMP,
                brand_appeared BOOLEAN,
                position TEXT,
                positive_keywords TEXT,
                negative_keywords TEXT,
                competitors_mentioned TEXT,
                references TEXT,
                screenshot_path TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        
        conn.commit()
        conn.close()
    
    def _save_to_database(self, result: GEOResult):
        """保存结果到数据库"""
        conn = sqlite3.connect('geo_monitor.db')
        cursor = conn.cursor()
        
        cursor.execute('''
            INSERT INTO geo_monitor_results 
            (question, ai_model, target_brand, ai_answer, query_time, 
             brand_appeared, position, positive_keywords, negative_keywords,
             competitors_mentioned, references, screenshot_path)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            result.question,
            result.ai_model,
            result.target_brand,
            result.ai_answer,
            result.query_time,
            result.brand_appeared,
            result.position,
            result.positive_keywords,
            result.negative_keywords,
            result.competitors_mentioned,
            result.references,
            result.screenshot_path
        ))
        
        conn.commit()
        conn.close()
    
    def search_baidu(self, question: str, target_brand: str) -> GEOResult:
        """
        百度搜索AI摘要抓取
        
        抓取内容：
        1. AI智能回答卡片
        2. 实时智能回复
        3. 传统搜索结果
        """
        print(f"🔍 查询百度: {question}")
        
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=True)
            page = browser.new_page()
            
            try:
                # 访问百度搜索
                encoded_question = question.replace(" ", "%20")
                url = f"https://www.baidu.com/s?wd={encoded_question}"
                page.goto(url, wait_until="networkidle", timeout=30000)
                
                # 等待AI摘要加载
                page.wait_for_timeout(3000)
                
                # 截图
                timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
                screenshot_file = f"{self.screenshot_dir}/baidu_{timestamp}.png"
                page.screenshot(path=screenshot_file, full_page=True)
                
                # 提取AI摘要内容
                ai_summary = self._extract_baidu_ai_summary(page)
                
                # 分析结果
                analysis = self._analyze_content(
                    ai_summary, 
                    target_brand,
                    page
                )
                
                browser.close()
                
                return GEOResult(
                    question=question,
                    ai_model="百度",
                    target_brand=target_brand,
                    ai_answer=ai_summary,
                    query_time=datetime.now(),
                    brand_appeared=analysis['brand_appeared'],
                    position=analysis['position'],
                    positive_keywords=analysis['positive_keywords'],
                    negative_keywords=analysis['negative_keywords'],
                    competitors_mentioned=analysis['competitors_mentioned'],
                    references=json.dumps(analysis['references'], ensure_ascii=False),
                    screenshot_path=screenshot_file
                )
                
            except Exception as e:
                browser.close()
                print(f"❌ 百度查询失败: {e}")
                return self._create_empty_result(question, "百度", target_brand, str(e))
    
    def _extract_baidu_ai_summary(self, page: Page) -> str:
        """提取百度AI摘要内容"""
        try:
            # 尝试获取AI智能回答
            ai_summary = page.evaluate('''() => {
                // 查找AI摘要区域
                const aiSection = document.querySelector('[class*="ai-"], [class*="AI-"]');
                if (aiSection) {
                    return aiSection.innerText;
                }
                
                // 备选：查找第一个结果摘要
                const firstResult = document.querySelector('.result, #content_left');
                if (firstResult) {
                    return firstResult.innerText.substring(0, 2000);
                }
                
                return "";
            }''')
            
            return ai_summary or "未找到AI摘要"
            
        except Exception as e:
            return f"提取失败: {e}"
    
    def _analyze_content(self, content: str, target_brand: str, page: Page) -> Dict:
        """
        分析内容，提取关键信息
        
        分析维度：
        1. 品牌是否出现
        2. 出现位置
        3. 正面/负面关键词
        4. 竞品提及
        5. 引用来源
        """
        result = {
            'brand_appeared': False,
            'position': '未出现',
            'positive_keywords': '',
            'negative_keywords': '',
            'competitors_mentioned': '',
            'references': []
        }
        
        if not content:
            return result
        
        # 1. 检测品牌出现
        brand_keywords = [target_brand, target_brand.replace('加速器', '')]
        for keyword in brand_keywords:
            if keyword in content:
                result['brand_appeared'] = True
                # 判断位置（简化版）
                index = content.find(keyword)
                if index < len(content) * 0.3:
                    result['position'] = '第1位'
                elif index < len(content) * 0.6:
                    result['position'] = '第2位'
                else:
                    result['position'] = '第3位及以后'
                break
        
        # 2. 提取正面关键词
        positive_words = ['推荐', '好用', '稳定', '快速', '专业', '免费', '优质', '首选']
        found_positive = [w for w in positive_words if w in content and target_brand in content]
        result['positive_keywords'] = ','.join(found_positive)
        
        # 3. 提取负面关键词
        negative_words = ['卡顿', '慢', '贵', '不好用', '问题', '错误']
        found_negative = [w for w in negative_words if w in content and target_brand in content]
        result['negative_keywords'] = ','.join(found_negative)
        
        # 4. 检测竞品
        competitors = ['迅游', '雷神', 'UU', '网易UU', '暴喵', 'biubiu']
        found_competitors = [c for c in competitors if c in content and c not in target_brand]
        result['competitors_mentioned'] = ','.join(found_competitors)
        
        # 5. 提取引用链接
        try:
            links = page.evaluate('''() => {
                const links = [];
                document.querySelectorAll('a[href*="baidu.com/link"]').forEach(a => {
                    links.push({
                        text: a.innerText.substring(0, 50),
                        url: a.href
                    });
                });
                return links.slice(0, 5);  // 只取前5个
            }''')
            result['references'] = links
        except:
            pass
        
        return result
    
    def _create_empty_result(self, question: str, ai_model: str, 
                            target_brand: str, error_msg: str) -> GEOResult:
        """创建空结果（查询失败时）"""
        return GEOResult(
            question=question,
            ai_model=ai_model,
            target_brand=target_brand,
            ai_answer=f"查询失败: {error_msg}",
            query_time=datetime.now(),
            brand_appeared=False,
            position='查询失败',
            positive_keywords='',
            negative_keywords='',
            competitors_mentioned='',
            references='[]',
            screenshot_path=''
        )
    
    def run_monitor(self):
        """执行监控任务"""
        print(f"\n{'='*60}")
        print(f"🚀 启动GEO监控 - {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        print(f"{'='*60}\n")
        
        # 只查询active状态的配置
        active_configs = self.config_df[self.config_df['status'] == 'active']
        
        for _, config in active_configs.iterrows():
            question = config['question']
            ai_model = config['ai_model']
            target_brand = config['target_brand']
            
            print(f"\n📋 监控任务: {question} ({ai_model})")
            
            # 根据平台选择查询方式
            if ai_model == "百度":
                result = self.search_baidu(question, target_brand)
            elif ai_model == "豆包":
                # result = self.search_doubao(question, target_brand)
                print("  ⏭️  豆包查询暂未实现")
                continue
            else:
                print(f"  ⚠️  不支持的平台: {ai_model}")
                continue
            
            # 保存结果
            self._save_to_database(result)
            self.results.append(asdict(result))
            
            # 打印结果
            status = "✅ 出现" if result.brand_appeared else "❌ 未出现"
            print(f"  {status} | 位置: {result.position}")
            if result.competitors_mentioned:
                print(f"  🏆 竞品: {result.competitors_mentioned}")
            
            # 间隔，避免请求过快
            time.sleep(2)
        
        # 导出Excel报告
        self._export_report()
        
        print(f"\n{'='*60}")
        print(f"✅ 监控完成 - 共查询 {len(self.results)} 条")
        print(f"{'='*60}\n")
    
    def _export_report(self):
        """导出监控报告"""
        if not self.results:
            return
        
        df = pd.DataFrame(self.results)
        
        # 添加统计列
        df['date'] = df['query_time'].apply(lambda x: x.strftime('%Y-%m-%d'))
        
        # 保存Excel
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        report_file = f"geo_report_{timestamp}.xlsx"
        
        with pd.ExcelWriter(report_file, engine='openpyxl') as writer:
            # 详细结果
            df.to_excel(writer, sheet_name='详细结果', index=False)
            
            # 统计摘要
            summary = {
                '总查询数': [len(df)],
                '品牌出现数': [df['brand_appeared'].sum()],
                '提及率': [f"{df['brand_appeared'].mean()*100:.1f}%"],
                '查询时间': [datetime.now().strftime('%Y-%m-%d %H:%M:%S')]
            }
            pd.DataFrame(summary).to_excel(writer, sheet_name='统计摘要', index=False)
        
        print(f"📊 报告已保存: {report_file}")


def main():
    """主函数"""
    monitor = GEOMonitor("geo_config.xlsx")
    monitor.run_monitor()


if __name__ == "__main__":
    main()
```
