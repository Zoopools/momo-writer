```python
def check_agfs_status():
    """检查 AGFS 状态，处理自愈场景"""
    import time
    import subprocess
    
    # 第一次检测
    result1 = subprocess.run(['nc', '-zv', '127.0.0.1', '1933'], capture_output=True)
    
    if result1.returncode != 0:
        # 端口不通，等待 5 秒
        time.sleep(5)
        
        # 第二次检测
        result2 = subprocess.run(['nc', '-zv', '127.0.0.1', '1933'], capture_output=True)
        
        if result2.returncode == 0:
            # 恢复，系统自愈完成
            return "🟢 系统自愈完成，AGFS 已恢复"
        else:
            # 仍异常，检查 LaunchAgent
            return "🔴 AGFS 异常，请检查 LaunchAgent 状态"
    else:
        return "🟢 AGFS 运行正常"
```
