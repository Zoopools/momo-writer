**执行次数计算：**
```
新媒体 Agent 每日执行：
- 数据监控：3 平台 × 1 次/天 = 3 执行
- 内容发布：3 平台 × 1 次/天 = 3 执行
```

**示例脚本：**
```python
#!/usr/bin/env python3
# 数据监控脚本

import sqlite3
import requests
from datetime import datetime

# 连接数据库
conn = sqlite3.connect('newmedia.db')
cursor = conn.cursor()

# 抓取抖音数据
def fetch_douyin_data():
    # API 调用
    response = requests.get('https://api.douyin.com/...')
    data = response.json()
    
    # 存入数据库
    cursor.execute('''
        INSERT INTO analytics (platform, content_id, views, likes, recorded_at)
        VALUES (?, ?, ?, ?, ?)
    ''', ('douyin', data['id'], data['views'], data['likes'], datetime.now()))
    
    conn.commit()

# 检查异常
def check_anomalies():
    cursor.execute('''
        SELECT * FROM analytics 
        WHERE recorded_at >= datetime('now', '-1 day')
        AND views < (SELECT AVG(views) * 0.5 FROM analytics WHERE platform = 'douyin')
    ''')
    anomalies = cursor.fetchall()
    
    if anomalies:
        # 发送飞书通知
        send_feishu_notification(anomalies)

# 主函数
if __name__ == '__main__':
    fetch_douyin_data()
    check_anomalies()
```
